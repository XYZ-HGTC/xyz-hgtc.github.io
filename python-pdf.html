实现PDF页数分割（12页一份）：

from PyPDF2 import PdfWriter, PdfReader


def split_pdf(input_file, output_prefix):
    input_pdf = PdfReader(open(input_file, "rb"))
    num_pages = len(input_pdf.pages)
    start_page = 0
    file_index = 1
    while start_page < num_pages:
        end_page = min(start_page + 12, num_pages)
        output_pdf = PdfWriter()
        for i in range(start_page, end_page):
            output_pdf.add_page(input_pdf.pages[i])
        output_path = f"{output_prefix}_{file_index}.pdf"
        with open(output_path, "wb") as output_file:
            output_pdf.write(output_file)
        start_page += 12
        file_index += 1
    print("complete")

import requests

  

user_input = input("请输入您的问题：\n")

  

# 设置API端点和密钥

api_url = "https://api.deepseek.com/v1/chat/completions"

api_key = " "

  

# 设置请求头

headers = {

    "Authorization": f"Bearer {api_key}",

    "Content-Type": "application/json"

}

  

# 设置请求体

data = {

    "model": "deepseek-chat",  # 选择模型

    "messages": [

        {"role": "system", "content": "You are a helpful assistant."},

        {"role": "user", "content": user_input}

    ]

}

  

# 发送POST请求

response = requests.post(api_url, headers=headers, json=data)

  

# 检查响应状态

if response.status_code == 200:

    # 解析响应内容

    result = response.json()

    print("DeepSeek回复:\n", result['choices'][0]['message']['content'])

else:

    print("请求失败，状态码:", response.status_code)

    print("错误信息:", response.text)
